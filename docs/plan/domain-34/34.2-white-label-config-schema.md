### 34.2 White-Label Config Schema

**Extension to `SiteConfig` for Enterprise tier:**

```typescript
// packages/config/src/types.ts (additions)

export interface WhiteLabelConfig {
  enabled: boolean;
  // Portal branding (replaces agency branding)
  portalName: string; // e.g. "Johns Plumbing Client Portal"
  portalLogoUrl?: string; // URL in Supabase Storage
  portalFaviconUrl?: string;
  portalPrimaryColor: string; // Portal primary CSS variable
  portalDomain?: string; // e.g. "portal.johnsplumbing.com"
  // Remove attribution
  hideAgencyBranding: boolean; // Removes "Powered by" footer
  hideSupportLink: boolean; // Hides support chat widget
  // Custom legal
  privacyPolicyUrl?: string;
  termsOfServiceUrl?: string;
  // Custom support
  supportEmail?: string;
  supportPhone?: string;
}
```
