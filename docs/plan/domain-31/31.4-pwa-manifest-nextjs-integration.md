### 31.4 PWA Manifest + Next.js Integration

**File:** `apps/*/src/app/manifest.ts`

```typescript
import type { MetadataRoute } from 'next';
import { headers } from 'next/headers';
import { resolveTenant } from '@repo/multi-tenant/resolve-tenant';
import { db } from '@repo/db';

export const dynamic = 'force-dynamic';

// Per-tenant PWA manifest â€” name, theme color, and icons from tenant config
export default async function manifest(): Promise<MetadataRoute.Manifest> {
  const headersList = await headers();
  const tenantContext = await resolveTenant({ headers: headersList } as any);

  // Default manifest if tenant not resolved
  if (!tenantContext) {
    return {
      name: 'Business Site',
      short_name: 'Business',
      theme_color: '#2563eb',
      background_color: '#ffffff',
      display: 'standalone',
      start_url: '/',
      icons: [
        { src: '/icon-192.png', sizes: '192x192', type: 'image/png' },
        { src: '/icon-512.png', sizes: '512x512', type: 'image/png' },
      ],
    };
  }

  const { data: tenant } = await db
    .from('tenants')
    .select('config')
    .eq('id', tenantContext.tenantId)
    .single();

  const config = tenant?.config as any;
  const identity = config?.identity ?? {};
  const primaryColor = config?.theme?.colors?.primary ?? '#2563eb';
  const logo = config?.assets?.logo;

  return {
    name: identity.siteName ?? 'Business Site',
    short_name: (identity.siteName ?? 'Business').slice(0, 12),
    description: identity.tagline,
    theme_color: primaryColor,
    background_color: '#ffffff',
    display: 'standalone',
    start_url: '/?source=pwa',
    scope: '/',
    orientation: 'portrait-primary',
    categories: ['business', 'productivity'],
    icons: [
      ...(logo
        ? [
            { src: logo, sizes: '192x192', type: 'image/png', purpose: 'any maskable' as const },
            { src: logo, sizes: '512x512', type: 'image/png', purpose: 'any' as const },
          ]
        : [
            { src: '/icon-192.png', sizes: '192x192', type: 'image/png' },
            { src: '/icon-512.png', sizes: '512x512', type: 'image/png' },
          ]),
    ],
    shortcuts: [
      {
        name: 'Contact Us',
        short_name: 'Contact',
        url: '/contact?source=pwa-shortcut',
        icons: [{ src: '/icon-contact.png', sizes: '96x96' }],
      },
      {
        name: 'Book Appointment',
        short_name: 'Book',
        url: '/book?source=pwa-shortcut',
        icons: [{ src: '/icon-book.png', sizes: '96x96' }],
      },
    ],
  };
}
```

```typescript
'use client';

import { useEffect, useState } from 'react';

export function OfflineBanner() {
  const [isOffline, setIsOffline] = useState(false);
  const [justReconnected, setJustReconnected] = useState(false);

  useEffect(() => {
    const handleOffline = () => {
      setIsOffline(true);
      setJustReconnected(false);
    };
    const handleOnline = () => {
      setIsOffline(false);
      setJustReconnected(true);
      setTimeout(() => setJustReconnected(false), 4000);
    };

    window.addEventListener('offline', handleOffline);
    window.addEventListener('online', handleOnline);

    return () => {
      window.removeEventListener('offline', handleOffline);
      window.removeEventListener('online', handleOnline);
    };
  }, []);

  if (!isOffline && !justReconnected) return null;

  return (
    <div
      role="status"
      aria-live="polite"
      className={`
        fixed top-0 left-0 right-0 z-[100] py-2 px-4 text-center text-sm font-medium
        transition-all duration-300
        ${isOffline ? 'bg-yellow-500 text-yellow-900' : 'bg-green-500 text-white'}
      `}
    >
      {isOffline
        ? 'ðŸ“¶ You\'re offline â€” forms will be sent when you reconnect'
        : 'âœ“ Back online â€” syncing any pending submissionsâ€¦'}
    </div>
  );
}
```
