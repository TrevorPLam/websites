// File: apps/*/src/app/api/webhooks/speed-insights/route.ts
// Vercel sends webhook when RES drops below threshold for 15+ minutes

import { NextRequest, NextResponse } from 'next/server';

interface SpeedInsightsWebhookPayload {
  event: 'score_degraded' | 'score_recovered';
  metric: string;
  score: number;
  threshold: number;
  url: string;
  projectId: string;
}

export async function POST(req: NextRequest) {
  const secret = req.headers.get('x-vercel-signature');
  if (secret !== process.env.SPEED_INSIGHTS_WEBHOOK_SECRET) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
  }

  const payload: SpeedInsightsWebhookPayload = await req.json();

  if (payload.event === 'score_degraded') {
    // Alert the dev team via Slack
    await fetch(process.env.SLACK_WEBHOOK_PERF_ALERTS!, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        text:
          `ðŸš¨ *Performance Regression Detected*\n` +
          `â€¢ Metric: \`${payload.metric}\`\n` +
          `â€¢ Score: ${payload.score} (threshold: ${payload.threshold})\n` +
          `â€¢ URL: ${payload.url}\n` +
          `â€¢ Action: Check Vercel Speed Insights dashboard`,
      }),
    });
  }

  return NextResponse.json({ received: true });
}
