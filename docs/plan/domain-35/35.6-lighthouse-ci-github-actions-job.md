### 35.6 Lighthouse CI GitHub Actions Job

**File:** `.github/workflows/lighthouse.yml`

```yaml
# ============================================================================
# LIGHTHOUSE CI â€” Performance Gate
# Runs on every PR against the Vercel preview deployment.
# Fails the merge if LCP/INP/CLS/score thresholds are not met.
# Reference: https://github.com/GoogleChrome/lighthouse-ci
# ============================================================================

name: Performance Gate (Lighthouse CI)

on:
  pull_request:
    branches: [main, staging]
    paths:
      # Only run when marketing site or shared packages change
      - 'apps/marketing/**'
      - 'packages/**'
      - '!packages/reports/**' # Report package changes don't affect marketing site perf
      - '!**/*.test.ts'
      - '!**/*.spec.ts'

jobs:
  lighthouse:
    name: Lighthouse Audit
    runs-on: ubuntu-latest
    timeout-minutes: 20

    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup pnpm
        uses: pnpm/action-setup@v3
        with:
          version: 9

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '22'
          cache: 'pnpm'

      # Wait for Vercel preview deployment to be ready
      - name: Wait for Vercel Preview
        uses: patrickedqvist/wait-for-vercel-preview@v1.3.1
        id: vercel-preview
        with:
          token: ${{ secrets.GITHUB_TOKEN }}
          max_timeout: 300 # 5 min max wait
          check_interval: 15
        env:
          VERCEL_TOKEN: ${{ secrets.VERCEL_TOKEN }}

      - name: Install LHCI
        run: pnpm add -g @lhci/cli@0.14.x

      # Collect, assert, and upload results
      - name: Run Lighthouse CI
        run: lhci autorun
        env:
          LHCI_GITHUB_APP_TOKEN: ${{ secrets.LHCI_GITHUB_APP_TOKEN }}
          LHCI_BASE_URL: ${{ steps.vercel-preview.outputs.url }}
          LHCI_BUILD_CONTEXT__CURRENT_BRANCH: ${{ github.head_ref }}
          LHCI_BUILD_CONTEXT__COMMIT: ${{ github.sha }}

      # Upload HTML report as artifact (accessible from Actions tab)
      - name: Upload Lighthouse Report
        uses: actions/upload-artifact@v4
        if: always()
        with:
          name: lighthouse-report-${{ github.sha }}
          path: .lighthouseci/
          retention-days: 30
```
